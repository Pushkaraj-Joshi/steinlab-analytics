def line_fit(x,y,mydataset1,mydataset2):
    
    mask = ~((x >= 0) & (x <= 20)) # remove outliers in size = 1
    x = x[mask]
    y = y[mask]
    
    plt.figure(figsize=(8,8))
    coefficients = np.polyfit(x, y, 1)
    polynomial = np.poly1d(coefficients)
    y_fit = polynomial(x)
    
    # Calculate R-squared
    y_mean = np.mean(y)
    ss_tot = np.sum((y - y_mean) ** 2)
    ss_res = np.sum((y - y_fit) ** 2)
    r = np.sqrt(1 - (ss_res / ss_tot))
    
    # Equation of the line
    equation = f'y = {coefficients[0]:.2f}x + {coefficients[1]:.2f}'

    plt.scatter(x, y, alpha=0.5, color='blue', label='Data points')
    plt.plot(x, y_fit, color='red', label='Fitted line')
    plt.xlabel(title[mydataset1])
    plt.ylabel(title[mydataset2])
    plt.legend()
    plt.text(0.1, 0.9, equation, transform=plt.gca().transAxes, fontsize=12, color='red')
    plt.text(0.1, 0.85, f'R = {r:.2f}', transform=plt.gca().transAxes, fontsize=12, color='red')
    plt.show()

def extract_regions_with_otsu(image, texture):
    """
    Extract two regions based on texture variance and original image pixel values:
    1. Low variance and low pixel values
    2. High variance and high pixel values

    Parameters:
        image (numpy.ndarray): Original grayscale image.
        texture (numpy.ndarray): Texture map (e.g., LBP or gradient variance).

    Returns:
        low_variance_region (numpy.ndarray): Low variance and low value region.
        high_variance_region (numpy.ndarray): High variance and high value region.
    """
    # Step 1: Compute gradient variance (local texture measure)
    kernel = np.ones((5, 5), dtype=np.float32) / 25  # Small kernel for local neighborhood
    local_mean = cv2.filter2D(texture, -1, kernel)
    local_variance = cv2.filter2D((texture - local_mean) ** 2, -1, kernel)

    # Step 2: Calculate the threshold for texture variance (Otsu's threshold)
    _, variance_thresholded = cv2.threshold(local_variance, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # Step 3: Threshold original image to find high and low pixel value regions (Otsu's threshold)
    _, image_thresholded = cv2.threshold(image, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)

    # Step 4: Extract low variance or low value region
    low_variance_region = np.where((variance_thresholded == 0) | (image_thresholded == 0), image, 0)

    # Step 5: Extract high variance or high value region
    high_variance_region = np.where((variance_thresholded == 255) | (image_thresholded == 255), image, 0)

    return low_variance_region, high_variance_region

def calculate_histogram_correlation(hist1, hist2):
    """
    Calculate histogram correlation using normalized cross-correlation.

    Parameters:
        hist1 (numpy.ndarray): Histogram of the first image.
        hist2 (numpy.ndarray): Histogram of the second image.

    Returns:
        correlation (float): The correlation coefficient between the two histograms.
    """
    correlation = np.corrcoef(hist1, hist2)[0, 1]
    return correlation

def analyze_images_advanced(image1, image2, mydataset1, mydataset2):
    """
    Analyze two images using advanced features like texture, and histogram distributions.
    Includes normalization and denoising before correlation analysis.

    Parameters:
        image1 (numpy.ndarray): First input image (reference for masking).
        image2 (numpy.ndarray): Second input image (to be mapped using the first image's masks).

    Returns:
        None
    """
    # Step 0: Preprocessing (Normalization and Denoising)
    image1 = cv2.normalize(image1, None, 0, 255, cv2.NORM_MINMAX).astype(np.uint8)
    image2 = cv2.normalize(image2, None, 0, 255, cv2.NORM_MINMAX).astype(np.uint8)
    image1 = cv2.GaussianBlur(image1, (3, 3), 0)
    image2 = cv2.GaussianBlur(image2, (3, 3), 0)

    # Normalization to bring the images into the range [0, 255]
    image1 = cv2.normalize(image1, None, 0, 255, cv2.NORM_MINMAX)
    image2 = cv2.normalize(image2, None, 0, 255, cv2.NORM_MINMAX)

    # Convert images to uint8
    image1 = np.uint8(image1)
    image2 = np.uint8(image2)

    # Step 1: Compute texture maps using Local Binary Pattern (LBP)
    texture1 = feature.local_binary_pattern(image1, P=8, R=1, method="uniform")
    texture2 = feature.local_binary_pattern(image2, P=8, R=1, method="uniform")

    # Convert texture maps to uint8
    texture1 = np.uint8(texture1)
    texture2 = np.uint8(texture2)

    # Step 2: Extract high and low variance regions for both textures using Otsu's threshold
    low_region1, high_region1 = extract_regions_with_otsu(image1,texture1)
    low_region2, high_region2 = extract_regions_with_otsu(image2,texture2)
    
    exclusive_low_region = np.where((low_region1 == 0) & (low_region2 > 0), low_region2, 0)
    exclusive_high_region = np.where((high_region1 > 0) & (high_region2 == 0), high_region1, 0)

    # Step 3: Calculate histogram correlations
    hist_image1 = cv2.calcHist([image1], [0], None, [256], [0, 256])
    hist_image2 = cv2.calcHist([image2], [0], None, [256], [0, 256])
    hist_low_region1 = cv2.calcHist([low_region1], [0], None, [256], [0, 256])
    hist_low_region2 = cv2.calcHist([low_region2], [0], None, [256], [0, 256])

    correlation_image_hist = calculate_histogram_correlation(hist_image1.flatten(), hist_image2.flatten())
    correlation_low_region_hist = calculate_histogram_correlation(hist_low_region1.flatten(), hist_low_region2.flatten())

    # Step 4: Visualization and Results
    fig, axes = plt.subplots(4, 3, figsize=(16, 11))

    # Plot original images
    axes[0, 0].imshow(image1, cmap="gray")
    axes[0, 0].set_title(f"Sizing = {sizing[mydataset1]}")

    axes[0, 1].imshow(image2, cmap="gray")
    axes[0, 1].set_title(f"Sizing = {sizing[mydataset2]}")

    # Plot histograms of original images on the same plot
    axes[0, 2].hist(image1.flatten(), bins=256, range=(0, 256), color="blue", alpha=0.7, label="Image 1")
    axes[0, 2].hist(image2.flatten(), bins=256, range=(0, 256), color="orange", alpha=0.7, label="Image 2")
    axes[0, 2].set_title(f"Histogram of Image 1 and Image 2 (Correlation: {correlation_image_hist:.4f})")
    axes[0, 2].legend()
    
    # Plot texture maps
    axes[1, 0].imshow(texture1, cmap="gray")
    axes[1, 0].set_title("Texture Map of Image 1")

    axes[1, 1].imshow(texture2, cmap="gray")
    axes[1, 1].set_title("Texture Map of Image 2")
    
    # Plot histograms of texture maps on the same plot
    axes[1, 2].hist(texture1.flatten(), bins=256, range=(0, 12), color="blue", alpha=0.7, label="Image 1")
    axes[1, 2].hist(texture2.flatten(), bins=256, range=(0, 12), color="orange", alpha=0.7, label="Image 2")
    axes[1, 2].set_title(f"Histogram of Textures of Image 1 and Image 2 (Correlation: {correlation_image_hist:.4f})")
    axes[1, 2].legend()
    
    # Plot the low and high variance regions for Image 1
    axes[2, 0].imshow(low_region1, cmap="gray")
    axes[2, 0].set_title("Biofilm Region Image 1")

    axes[2, 1].imshow(low_region2, cmap="gray")
    axes[2, 1].set_title("Biofilm Region Image 2")
    
    axes[2, 2].imshow(exclusive_low_region, cmap="gray")
    axes[2, 2].set_title("Diff Biofilm Region (Image 1 over Image 2)")
    
    # Plot the low and high variance regions for Image 1
    axes[3, 0].imshow(high_region1, cmap="gray")
    axes[3, 0].set_title("Medium Region (Image 1)")

    axes[3, 1].imshow(high_region2, cmap="gray")
    axes[3, 1].set_title("Medium Region (Image 2)")
    
    axes[3, 2].imshow(exclusive_high_region, cmap="gray")

def classify_image(image, K=3, min_region_size=4000):
    """
    Classifies a grayscale image into K continuous parts, ensuring small isolated regions 
    are merged with the surrounding region.
    
    Parameters:
        image (numpy array): Grayscale image (2D NumPy array).
        K (int): Number of clusters for segmentation (default is 3).
        min_region_size (int): Minimum size for a region to remain separate.
        
    Returns:
        segmented_image (numpy array): Image with classified continuous regions.
    """
    # Ensure the image is in uint8 format
    image = np.uint8(image)

    # **Step 1: Remove outliers using median filtering**
    cleaned_image = cv2.medianBlur(image, 3)

    # **Step 2: Apply K-means clustering**
    Z = cleaned_image.reshape((-1, 1)).astype(np.float32)
    criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 10, 1.0)
    _, labels, centers = cv2.kmeans(Z, K, None, criteria, 10, cv2.KMEANS_RANDOM_CENTERS)

    # Reshape labels back to original image size
    segmented_image = labels.reshape(cleaned_image.shape)

    # **Step 3: Ensure continuous regions using connected component analysis**
    for k in range(K):
        mask = (segmented_image == k).astype(np.uint8)  # Binary mask for class k
        
        # Label connected components
        labeled_array, num_features = label(mask)

        # Loop through connected components
        for i in range(1, num_features + 1):
            region_size = np.sum(labeled_array == i)

            # If the region is too small, merge it into the largest neighboring region
            if region_size < min_region_size:
                # Find the most common surrounding label
                dilated_mask = cv2.dilate((labeled_array == i).astype(np.uint8), 
                                          cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5,5)))

                neighbor_labels = segmented_image[dilated_mask.astype(bool)]
                most_common_label = np.bincount(neighbor_labels[neighbor_labels != k]).argmax()

                # Reassign the small region to the most common surrounding label
                segmented_image[labeled_array == i] = most_common_label

    # **Step 4: Apply morphological operations to each class separately**
    kernel = np.ones((3,3), np.uint8)
    segmented_image_cleaned = np.zeros_like(segmented_image)

    for k in range(K):
        region_mask = (segmented_image == k).astype(np.uint8)  # Convert class to binary
        region_mask = cv2.morphologyEx(region_mask, cv2.MORPH_CLOSE, kernel)  # Apply closing
        segmented_image_cleaned[region_mask == 1] = k  # Restore class label

    # **Step 5: Color the regions based on their labels (Blue, Red, Green)**

    colored_image = np.zeros((segmented_image_cleaned.shape[0], segmented_image_cleaned.shape[1], 3), dtype=np.uint8)

    # Assign colors: Blue (Region 1), Red (Region 2), Green (Region 3)
    region_means = []
    for k in range(K):
        region_mask = (segmented_image_cleaned == k)
        region_values = image[region_mask]
        region_means.append(np.mean(region_values))
    size = [0]*K
    for k in range(K):
        if np.mean(image[segmented_image_cleaned == k]) == max(region_means):  
            colored_image[segmented_image_cleaned == k] = [255, 0, 0]   # Blue
            print('Size of Region 1 = %u'%len(colored_image[segmented_image_cleaned == k]))
            size[0] = len(colored_image[segmented_image_cleaned == k])
        elif np.mean(image[segmented_image_cleaned == k]) == min(region_means):  
            colored_image[segmented_image_cleaned == k] = [0, 0, 255]   # Red
            print('Size of Region 2 = %u'%len(colored_image[segmented_image_cleaned == k]))
            size[1] = len(colored_image[segmented_image_cleaned == k])
        else:
            colored_image[segmented_image_cleaned == k] = [0, 255, 0]   # Green

    # **Step 6: Display results**
    plt.figure(figsize=(12, 6))

    plt.subplot(1, 2, 1)
    plot_capacitance_image(plt,     # where to plot
                           image,  # image (fF)
                           title='', # title
                           color_stds=(6,6),) # std color ranges (vmin,vmax)

    plt.subplot(1, 2, 2)
    plt.title("Segmented Image (Continuous Regions)")
    plt.imshow(cv2.flip(colored_image, 0))
    plt.axis('off')
    
    plt.show()

    return colored_image,size
